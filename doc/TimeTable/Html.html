def initialize(filename = "timetable.html", html = DEFAULT_RHTML, input_type = "f")
  @filename = filename
  if input_type == "f"
    @html = ERB.new(File.new(html).read, nil, "%")
  elsif input_type == "s"
    @html = ERB.new(html.gsub(/^  /, ''), nil, "%")
  end
end
