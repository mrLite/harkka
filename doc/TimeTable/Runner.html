def run
  if @argv.empty?
    puts "html-tiedosto: timetable.html"
    html_file = Html.new
    html_file.produce_html
  elsif @argv and !@argv.empty?
    unless @argv.include? "-pdf" or @argv.include? "-html"
      puts USAGE
      Kernel.exit
    else
      output_html
      output_pdf
    end
  end
end
